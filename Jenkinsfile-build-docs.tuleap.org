pipeline {
    agent {
        label 'docker'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Build and publish image') {
            steps {
                script {
                    def image = docker.build('tuleap-documentation.docs.tuleap.org', "-f Dockerfile-docs.tuleap.org .")

                    docker.withRegistry('https://nexus.enalean.com:22000', 'ci-write') {
                        image.push()
                    }
                }
            }
        }
        stage('Deploy') {
            steps {
                sshagent(['deploy-documentation-web01.enalean.com']) {
                    sh 'ssh jenkins@web01.enalean.com ./update_docs.tuleap.org.sh'
                }
            }
        }
    }
}
